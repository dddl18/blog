                                                # 原型

                                                原型对象：javascript中，每一个函数在创建的时候，浏览器都会为这个函数创建一个对象，叫做原型对象 prototype

                                                原型属性：如果我们使用构造函数来创建一个对象时，创建出的这个对象具有原型属性，可以通过__proto__读取原型属性

                                                作用：我们可以在原型对象中添加属性，这样所有的对象都会拥有这个属性，实际上原型属性就是为了解决js继承问题。

                                                //构造函数(具有原型对象)
function Student(){
    console.log(123);
}
//获取原型对象
Student.prototype;
//给原型对象加属性和方法
Student.prototype.age =18;
Student.prototype.say =function(){
    console.log('我是say方法');
}



//通过构造函数创建对象(具有原型属性)
var xiaowang = new Student();
var xiaoli = new Student();
//获取原型属性
xiaowang.__proto__;
//当原型对象中有属性和方法时可以通过原型属性获取
xiaowang.__proto__.age;
xiaowang.__proto__.say();

                                                # 闭包

                                                由于函数存在作用域的问题，函数的外部无法获取到函数内部的成员。

                                                在一个函数内部，再写一个函数，然后由外层的函数将内层函数return出来，内层函数使用外层函数的变量，产生了闭包，如果想要使用外层函数中的变量，内部函数将变量返回出来即可。

                                                原理：由于函数自己拥有自己的作用域，每一个函数在运行开始到运行结束，创建变量，然后释放变量。

                                                在闭包中，由于内部函数使用了外部函数的变量，导致外部函数执行完没有办法回收他的变量（因为内部的函数还在使用）,此时该变量就会一直得不到释放。

                                                优点：在内部将变量返回，外部就可以获取

                                                缺点：由于变量一直得不到释放，所以会损害程序性能

                                                function outfunc()
{   
    var number=100;				//number外层函数中声明的 
    var infunc= function (){
        number++;				//内存的函数使用了外部函数的变量
        console.log('child');
        return number;			//将外部函数变量返回
    }
    return func;				//外部函数返回内部函数名
}
//		 得到内部函数名 执行内部函数
var value = outfunc()();

                                                # apply和call

                                                参数1：传到函数中的对象 参数2：向函数传递的参数

                                                区别：apply传递参数是以数组形式传递的，而call是以对象来传递，多个对象用逗号隔开即可

                                                作用：1.可以将任意的对象传入到函数的this中 2.实现继承

                                                作用1：传入任意对象

    name:'张三',
    age:20,
    say:function(){
        console.log(this);
    }
}
function test(name,age)
{
    console.log(name);
    console.log(age);
    console.log(this);
}
test.apply(student,['李四',20]);
test.call(student,{name:'王五',age:20});

                                                作用2：继承

                                                var animal = {
    run:function(){
        console.log(this.name + '都会飞');
    },
    eat:function(){
        console.log(this.name+'会吃')
    }
}
function Cat() {
    //this是对象本身
    this.name = '猫';
    this.run = function () {
        //通过call方法让对象的run方法为调用animal对象的run方法
        Animal.run.call(this)
    }
    this.eat = function(){
        //通过apply方法让对象的eat方法为调用animal对象的run方法
        Animal.eat.apply(this)
    }
}

function Dog() {
    this.name = "狗";
    this.run = function () {
        Animal.run.call(this);
    }
    this.eat = function(){
        Animal.eat.apply(this)
    }
}
var c = new Cat();
c.run();
c.eat();
var d = new Dog();
d.run();
d.eat();

                                                # Promise异步包装器

                                                不使用异步包装器的话，如果需要调用多个接口获取数据到一个变量中，则需要进行异步的嵌套，形成回调地狱（异步套异步），代码不优雅也不易理解维护。

                                                Promise三种状态：

                                                1. pending:等待状态，比如正在进行网络请求，或者定时器没有到时间。

                                                2. fulfill:满足状态，当我们主动回调了resolve时，就处于该状态，并且会回调.then()

                                                3. reject:拒绝状态，当我们主动回调了reject时，就处于该状态，并且会回调.catch()

                                                语法：new Promise((resolve,reject)=>{异步代码}).then(res=>{}).catch(err=>{}).finally();

                                                resolve：异步执行成功需要执行的函数

                                                reject：异步执行失败需要执行的函数

                                                标准使用

                                                new Promise((resolve,reject)=>{
    $.ajax({
        url:'',//路径
        type:'get',//设置访问方式
        //成功的回调函数
        success:res=>{
            resolve(res);
        },
        //失败的回调函数
        error:err=>{
            reject(err);
        }
    });
}).then(res=>{
    //执行成功后的逻辑
}).catch(err=>{
    //执行失败后的逻辑
})

                                                省略reject

                                                new Promise(resolve=>{
    //异步函数 
    $.get('jd.json',()=>{
        resolve();
    })
}).then(()=>{
    //执行成功后的逻辑
})

                                                resolve存在参数

                                                new Promise(resolve=>{
    //异步函数 
    $.get('jd.json',res=>{
        resolve(res);
    })
}).then(res=>{
    //执行成功后的逻辑
})

                                                Promise的链式调用

                                                let list = [];
new Promise((resolve,reject)=>{
    $.get('jd.json',res=>{
        resolve(res);
    })
}).then(res=>{
    list = list.concat(res);
    //返回一个新的Promise对象(taobao)
    return new Promise((resolve)=>{
        $.get('taobao.json',res=>{
        	resolve(res);
    	})
    })
}).then(tb=>{
    //新的Promise对象(taobao)的成功后的逻辑
    list = list.concat(tb);
    //返回一个新的Promise对象(suning)
    return new Promise((resolve)=>{
        $.get('suning.json',res=>{
        	resolve(res);
    	})
    })
}).then(sn=>{
    //新的Promise对象(suning)的成功后的逻辑
    list = list.concat(sn);
})

                                                new Promise((resolve,reject)=>{
    $.get('jd.json',res=>{
        resolve(res);
    })
}).then(res=>{
    return Promise.resolve(res+88)	//简化：return res+88

    //data:res+88

})

                                                Promise.all：当我们同时发送了N个异步请求后，我们并不能准确的知道哪一个请求先完成，哪一个是后完成的。此时我们可以使用promise中的all方法，all方法会等到所有的异步函数都执行成功后，给你进行回调。

                                                Promise.all([
    new Promise(resolve=>{
        setTimeout(() => {
            resolve(1);
        }, 1000);
    }),
    new Promise(resolve=>{
        setTimeout(() => {
            resolve(2);
        }, 3000);
    }),
    new Promise(resolve=>{
        setTimeout(() => {
            resolve(3);
        }, 2000);
    }),
]).then(data=>{
    //data为所有异步函数的结果组成的数组，如果无参则为数组里存undefined
    console.log(data);
})

                                                # ES7-Aysnc

                                                异步函数是使用async关键字声明的函数

                                                async函数是AsyncFunction构造函数的实例， 并且其中允许使用await关键字。

                                                async和await关键字让我们可以用一种更简洁的方式写出基于Promise的异步行为，而无需刻意地链式调用promise。

                                                使用场景：需要等待一个异步代码执行完成后再执行后续的代码。

                                                返回值：promise（fulfill状态的promise该对象为回调resolve/reject中的参数值）

                                                使用：

                                                1. 执行的方法中包含异步代码
                                                2. 希望异步调用按照顺序执行
                                                3. 异步代码必须用Promise包装并返回Promise对象
                                                // 执行的方法
function getData(url){
    //将Promise返回出去
    return new Promise(resolve=>{
        $.get(url,res=>{
            resolve(res);
        })
    })
}

// 异步函数
async function donow(){
    //		等待异步代码执行完毕
    let a = await getData('data.json');
    let b = await getData('data2.json');
    let c = await getData('data3.json');
    console.log(a,b,c);
}

donow();
