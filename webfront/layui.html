        # 弹出层

        层类型参数(type)：

        • 0：信息框（默认）
        • 1：页面层
        • 2：iframe层，content为子网页地址
        • 3： 加载层
        • 4： tips层， content: ['内容', '#id'] //数组第二项即吸附元素选择器或者DOM

        宽高 （area）： ['500px', '300px']

        layui.use('layer',function(){
    var layer = layui.layer;
    
    layer.open({
        type: 2,
        content: '/home/index'
    });
})

        <!--引css和js-->
<link href="~/Scripts/layui/css/layui.css" rel="stylesheet" />

<script>
    layui.use('table', function () {
        //声明一个table
        var table = layui.table;
		//渲染表格
        table.render({
            //指定渲染哪个表格
            elem: '#test'
            //数据的后端接口
            , url: '/student/Query'
            //是否使用分页
            , page: true
            //分页默认几个一页
            , limit: 5
            //下拉框可选的每页个数
            , limits: [5,10,15,20]
            , cellMinWidth: 80 //全局定义常规单元格的最小宽度，layui 2.2.1 新增
            //列数据设置（表头参数）
            , cols: [[
                { field: 'Id', title: 'ID', sort: true }
                , { field: 'Name', title: '姓名' }
                , { field: 'ProvinceID', title: '省份', sort: true }
                , { field: 'sex', title: '性别' }
                , { field: 'img_url', title: '图片' }
            ]]
        });
    });
</script>

<script>
    layui.use('table', function () {
        var table = layui.table;
        table.render({
            elem: '#test'
            , url: '/student/Query'
            , page: true
            , limit: 5
            , limits: [5,10,15,20]
            , cellMinWidth: 80
            , cols: [[
                { field: 'Id', title: 'ID', sort: true }
                , { field: 'Name', title: '姓名' }
                , { field: 'ProvinceID', title: '省份', sort: true }
                									//自定义模板 使用法二：函数
                , { field: 'sex', title: '性别', templet: getGendar }
                									//自定义模板 使用法一：绑定模板选择器
                , { field: 'img_url', title: '图片' , templet:'#img'}
            ]]
        });
    });
    function getGendar(d) {
            if (d.sex == 1) {
                return "男";
            } else {
                return "女";
            }
        }
</script>

        基础参数

        page ： 开启分页（默认：false）。 支持传入一个对象，里面可包含 laypage (opens new window) 组件所有支持的参数（jump、elem除外）

        limit ： 每页显示的条数（默认 10）

        limits ： 每页条数的选择项，默认：[10,20,30,40,50,60,70,80,90]

        表头参数

        field ： 设定字段名 （数据的字段名）

        title ： 设定标题名称 （页面显示的字段名）

        width ： 设定列宽，若不填写，则自动分配；若填写，则支持值为：数字、百分比。

        sort ： 是否允许排序（默认：false）

        align ： 单元格排列方式。可选值有：left（默认）、center（居中）、right（居右）

        templet ： 自定义列模板，模板遵循 laytpl (opens new window) 语法

        tips：如果使用分页需要在接口接收page和limit参数，通过该参数进行页面绑定数据的限制

<script>
    layui.use(['table'], function () {
        var table = layui.table;
        //渲染
        ...
        
        //搜索功能
        //使用layui里面的jQuery
        var $ = layui.$, active = {
            //重新加载表格
            reload: function () {
                //获取搜索框元素
                var search_name = $('#searchName');
                //执行重载		表格id
                table.reload('tb', {
                    page: {
                        curr: 1 //重新从第 1 页开始
                    }
                    // querystring传递的参数
                    , where: {
                        name: search_name.val()
                    }
                });
            }
        };
        //给搜索按钮添加点击监视
        $('.demoTable .layui-btn').on('click', function () {
            var type = $(this).data('type');
            active[type] ? active[type].call(this) : '';
        });
</script>

<script>
    layui.use(['laydate','form'], function () {
        var laydate = layui.laydate;
        var form = layui.form;
        //常规用法
        laydate.render({
            elem: '#test1'
        });

        //判断是添加还是更新,是更新则赋值
        if ($("#hfid").val() != "") {
            $.post("/student/layuiGetModel/" + $("#hfid").val(), function (res) {
                form.val('example', res);
            })
        }

        //按钮监控
        $("#btnSubmit").click(function () {
            //表单取值
            var data = form.val('example');
            var url = "";
            if (data.id == "") {
                data.id = 0;
                url = "/student/layuiAdd";
            } else {
                url = "/student/layuiUpdate"
            }
            $.post(url, { data: JSON.stringify(data) }, function (res) {
                console.log(res);
                if (res.code == "200") {
                    console.log(2)
                    layer.msg(res.msg);
                    //刷新父页面
                    parent.table.reload("tb");
                    //关闭弹出层
                    var index = parent.layer.getFrameIndex(window.name);
                    //从父页面上把layer关闭
                    parent.layer.close(index);
                }
            })
        })
    });

</script>

<div class="layui-upload">

<script>
    layui.use(['upload', 'element', 'layer'], function () {
        var $ = layui.jquery
        , upload = layui.upload
        , element = layui.element
        , layer = layui.layer;
        //常规使用 - 普通图片上传
        var uploadInst = upload.render({
            elem: '#btnUpload'
            , url: '/student/fileUpload' //此处用的是第三方的 http 请求演示，实际使用时改成您自己的上传接口即可。
            , before: function (obj) {
                //预读本地文件示例，不支持ie8
                obj.preview(function (index, file, result) {
                    $('#demo1').attr('src', result); //图片链接（base64）
                });

                element.progress('demo', '0%'); //进度条复位
                layer.msg('上传中', { icon: 16, time: 0 });
            }
            , done: function (res) {
                //如果上传失败
                if (res.code > 0) {
                    return layer.msg('上传失败');
                }
                //上传成功的一些操作
                //……
                $('#demoText').html(''); //置空上传失败的状态
            }
            , error: function () {
                //演示失败状态，并实现重传
                var demoText = $('#demoText');
                demoText.html('<span style="color: #FF5722;">上传失败</span> <a class="layui-btn layui-btn-xs demo-reload">重试</a>');
                demoText.find('.demo-reload').on('click', function () {
                    uploadInst.upload();
                });
            }
            //进度条
            , progress: function (n, elem, e) {
                element.progress('demo', n + '%'); //可配合 layui 进度条元素使用
                if (n == 100) {
                    layer.msg('上传完毕', { icon: 1 });
                }
            }
        });
    })
</script>
